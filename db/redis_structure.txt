# Redis Key Structure for News Aggregator

# Cached post data
# JSON поста
SET cache:post:<post_id> "<JSON>" EX 300

# User feed (sorted by time or score)
# Лента постов пользователя (post_id + время или рейтинг)
ZADD feed:user:<user_id> <timestamp> <post_id>
ZREVRANGE feed:user:<user_id> 0 49

# View counters
# Количество просмотров поста
INCR views:post:<post_id>

# Like counters
# Количество лайков поста
INCR likes:post:<post_id>

# Background task queue
# Очередь фоновых задач
LPUSH queue:fetch "<JSON>"

# User session
# Сессия пользователя
SET session:<token> <user_id> EX 3600
